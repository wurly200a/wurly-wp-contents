#+TITLE: Kubernetesクラスターのアップグレード (v1.33.9 -> v1.34.5)

* 概要

* clientのアップグレード

#+begin_src bash
echo "deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://pkgs.k8s.io/core:/stable:/v1.34/deb/ /" \
  | sudo tee /etc/apt/sources.list.d/kubernetes.list
sudo apt update
sudo apt upgrade kubectl
#+end_src

* apt repoのアップデート

全ノードで実行します。

#+begin_src bash
echo "deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://pkgs.k8s.io/core:/stable:/v1.34/deb/ /" \
  | sudo tee /etc/apt/sources.list.d/kubernetes.list
sudo apt-get update
#+end_src

* アップグレード from v1.33.9 to v1.34.5

** 0) まず ctrl1 で etcd snapshot

※rootシェルに入るので、exitの実行を忘れないようにすること

#+begin_src bash
sudo -i
ETCDCTL_API=3 etcdctl snapshot save /root/etcd-$(date +%F-%H%M)-pre134.db \
  --endpoints=https://127.0.0.1:2379 \
  --cacert=/etc/kubernetes/pki/etcd/ca.crt \
  --cert=/etc/kubernetes/pki/etcd/server.crt \
  --key=/etc/kubernetes/pki/etcd/server.key
exit
#+end_src

** 1) ctrl1（最初の1台だけ “apply”）

sudo systemctl restart kubelet
#+end_src

** 2) ctrl2 / ctrl3（各ノード、順番に）

※ここは apply じゃなくて upgrade node

#+begin_src bash
sudo apt-get update
sudo apt-get install -y kubeadm=1.34.5-1.1
sudo kubeadm config images pull

sudo kubeadm upgrade node
sudo apt-get install -y kubelet=1.34.5-1.1 kubectl=1.34.5-1.1
sudo systemctl daemon-reload
sudo systemctl restart kubelet
#+end_src
