#+TITLE: Rook/Ceph v1.15 から v1.16 さらに v1.17 へのアップグレード手順（Helm版）

* 基本

** helmでデプロイされたバージョンの確認

#+begin_src bash
helm list -n rook-ceph
#+end_src

** クラスター HEALTH の確認

#+begin_src bash
kubectl -n rook-ceph get cephcluster
#+end_src

** rook-ceph-operator podの確認

#+begin_src bash
kubectl -n rook-ceph get pods -l app=rook-ceph-operator
#+end_src

** Deploymentのバージョン確認(単発)

#+begin_src bash
kubectl -n rook-ceph get deployments \
  -l rook_cluster=rook-ceph \
  -o jsonpath='{range .items[*]}{.metadata.name}{"  req/upd/avl: "}{.spec.replicas}{"/"}{.status.updatedReplicas}{"/"}{.status.readyReplicas}{"  rook-version="}{.metadata.labels.rook-version}{"\n"}{end}'
#+end_src

** Deploymentのバージョン確認(watch)

#+begin_src shell
watch --exec kubectl -n rook-ceph get deployments \
  -l rook_cluster=rook-ceph \
  -o jsonpath='{range .items[*]}{.metadata.name}{"  req/upd/avl: "}{.spec.replicas}{"/"}{.status.updatedReplicas}{"/"}{.status.readyReplicas}{"  rook-version="}{.metadata.labels.rook-version}{"\n"}{end}'
#+end_src

** Deploy(Upgrade)

#+begin_src bash
helm upgrade rook-ceph rook-release/rook-ceph \
  --namespace rook-ceph \
  --version v1.x.y
#+end_src

#+begin_src bash
helm upgrade rook-ceph-cluster rook-release/rook-ceph-cluster \
  --namespace rook-ceph \
  --set operatorNamespace=rook-ceph \
  -f cluster-values.yaml \
  --version v1.x.y
#+end_src

** cluster-values.yaml

#+begin_src yaml
cephClusterSpec:
  storage:
    nodes:
    - devices:
      - name: /dev/nvme0n1p3
      name: k8s-worker1
    - devices:
      - name: /dev/nvme0n1p3
      name: k8s-worker2
    - devices:
      - name: /dev/nvme0n1p3
      name: k8s-worker3
    useAllDevices: false
    useAllNodes: false
#+end_src

* Helm repo の更新

#+begin_src shell
helm repo update
#+end_src

利用可能な v1.x チャートのバージョンを確認します。

#+begin_src shell
curl -s https://charts.rook.io/release/index.yaml | grep "version: v1.x"
#+end_src

最新の v1.x.y をデプロイします。

* Rook/Ceph v1.15 to v1.16

- https://rook.io/docs/rook/v1.16/Upgrade/rook-upgrade/

#+begin_src bash
# 事前確認
kubectl -n rook-ceph get cephcluster
kubectl -n rook-ceph get pods | grep holder  # 何も出ないこと
#+end_src

#+begin_src bash
# バージョン確認
helm repo update
curl -s https://charts.rook.io/release/index.yaml | grep "version: v1.16"
#+end_src

#+begin_src bash
helm upgrade rook-ceph rook-release/rook-ceph \
  --namespace rook-ceph \
  --reset-values \
  --version v1.16.9
#+end_src

"Deploymentのバージョン確認(watch)"で全てバージョンが上がったことを確認して次に進む

#+begin_src bash
helm upgrade rook-ceph-cluster rook-release/rook-ceph-cluster \
  --namespace rook-ceph \
  --set operatorNamespace=rook-ceph \
  -f cluster-values.yaml \
  --version v1.16.9
#+end_src
